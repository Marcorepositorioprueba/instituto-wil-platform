function doPost(e) {
  try {
    const spreadsheet = SpreadsheetApp.getActiveSpreadsheet();
    const sheetName = "Respuestas de formulario 1";
    const sheet = spreadsheet.getSheetByName(sheetName);

    // --- DEBUGGING --- 
    // Registra el ID y nombre del archivo y la pestaña objetivo.
    console.log(`Intentando escribir en ARCHIVO: '${spreadsheet.getName()}' (ID: ${spreadsheet.getId()})`);
    console.log(`Buscando PESTAÑA con nombre: '${sheetName}'`);
    
    if (sheet) {
      console.log("PESTAÑA encontrada con éxito.");
    } else {
      console.error("¡ERROR! PESTAÑA NO ENCONTRADA. Verifica que el nombre sea idéntico.");
      throw new Error(`La pestaña '${sheetName}' no fue encontrada.`);
    }
    // --- FIN DEBUGGING ---

    const data = JSON.parse(e.postData.contents);

    sheet.appendRow([
      new Date(),
      data.nombreMaestro,
      data.horaEntrada,
      data.horaSalida,
      data.fecha,
      data.gruposAtendidos
    ]);

    return ContentService
      .createTextOutput(JSON.stringify({ "status": "success", "message": "Fila añadida." }))
      .setMimeType(ContentService.MimeType.JSON);

  } catch (error) {
    console.error("Error en doPost: " + error.toString());
    return ContentService
      .createTextOutput(JSON.stringify({ "status": "error", "message": error.toString() }))
      .setMimeType(ContentService.MimeType.JSON);
  }
}
